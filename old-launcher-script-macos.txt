#!/bin/zsh

APP_ROOT="$(cd "$(dirname "$0")/.." && pwd)"
APP_FILES="$APP_ROOT/Resources/app"
APP_HTML="$APP_FILES/grant-radar-matching.html"
HOST="127.0.0.1"
PORT="8765"
APP_URL="http://$HOST:$PORT/grant-radar-matching.html"
LOG_DIR="$HOME/Library/Logs/BANDI-SCANNER"
LOG_FILE="$LOG_DIR/app.log"

mkdir -p "$LOG_DIR"

if ! command -v node >/dev/null 2>&1; then
  open "$APP_HTML"
  exit 0
fi

if ! curl -fsS --max-time 2 "$APP_URL" >/dev/null 2>&1; then
  cd "$APP_FILES" || exit 1
  HOST="$HOST" PORT="$PORT" nohup node "$APP_FILES/grant-radar-server.mjs" >> "$LOG_FILE" 2>&1 &
  sleep 2
fi

if curl -fsS --max-time 2 "$APP_URL" >/dev/null 2>&1; then
  open "$APP_URL"
else
  open "$APP_HTML"
fi
exit 0
